```bash
$ sudo blkid
$ sudo mdadm --create --verbose /dev/md0 --level=0 --raid-devices=2 /dev/sda1 /dev/sdb1
$ sudo cat /proc/mdstat
$ sudo mkfs.ext4 /dev/md0
$ sudo mkdir -p /mnt/raid0
$ sudo mount /dev/md0 /mnt/raid0
```
